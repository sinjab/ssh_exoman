**Common pattern for all handlers:**
```typescript
import type { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { SchemaName } from "../schemas/schema-file";
import { resultToMcpResponse } from "../test-utils";
// Import service layer functions

interface ToolDeps {
  processManager: ProcessManager;
  config: AppConfig;
  logger: typeof logger;
}

export function registerToolName(server: McpServer, deps: ToolDeps) {
  server.registerTool(
    "tool_name",
    { title: "Tool Title", description: "Description", inputSchema: SchemaName },
    async (params) => {
      try {
        const result = await serviceFunction(params, deps);
        return resultToMcpResponse(result);
      } catch (error) {
        return resultToMcpResponse({
          success: false,
          error: { code: "INTERNAL_ERROR", message: error.message }
        });
      }
    }
  );
}
```

import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { loadConfig } from "./config";
import { logger } from "./structured-logger";
import { ProcessManager } from "./ssh/process-manager";
import { registerExecuteCommand } from "./tools/execute";
import { registerGetOutput } from "./tools/output";
import { registerGetStatus } from "./tools/status";
import { registerKillCommand } from "./tools/kill";
import { registerSecurityInfo } from "./tools/security-info";

export function createServer(): McpServer {
  const server = new McpServer({
    name: "ssh-exoman",
    version: "1.0.0",
  });

  const config = loadConfig();
  const processManager = new ProcessManager();

  const deps = { processManager, config, logger };

  // Register all tools
  registerExecuteCommand(server, deps);
  registerGetOutput(server, deps);
  registerGetStatus(server, deps);
  registerKillCommand(server, deps);
  registerSecurityInfo(server, deps);

  return server;
}
```

#!/usr/bin/env node
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
import { createServer } from "./server.js";
import { logger } from "./structured-logger.js";

async function main() {
  logger.info("Starting SSH Exoman MCP server...");

  const server = createServer();
  const transport = new StdioServerTransport();

  await server.connect(transport);

  logger.info("Server connected via stdio transport");
}

  process.exit(1);
});
```
